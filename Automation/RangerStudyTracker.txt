	// normaly doesn't excudes anything after first word
	"/selectitem \"whatzit <#" cknum "\" \r"

	set lastwordinselec - 1

				message "* no whatzit found skipping"

				message "* not data skipping_1"

							end if

// "5/4/26 6:55:51a You slaughtered a Rat. \ \" <-- full line, this might be a bug from the client.
// "Rat. \" is one word might be bigger then this. there are 6 words, you can only see 5
// finds if any dataset maches the first word at any part of the dataset, if it does check again for exact match.
// this will also add to kill number, into temp memory.
// dev messages will be removed later
dswordfilter
{
set nameck 1
//message "scan check"
label top
if dataset[nameck] != "Nothing"
	if dataset[nameck] != ""
	//message "checking"
		set addone dataset[nameck]
		set secondlastword checklastword
		set secondlastword - 1
			if logck.word[secondlastword] >= "."
				set allbutoneword logck.num_words
					setglobal badnam logck.word[secondlastword]
						call scramble
						//message "recall from scramble:" nama "!!! name one resaved data: " addone.word[0] ""
						if allbutoneword == 6
							if nama == addone.word[0]
								set matchedname addone.word[0]
								goto matched
							else
								goto cknext
							end if
						else if allbutoneword == 7
							if nama == addone.word[1]
								set matchedname addone.word[1]
								goto matched
							else
								goto cknext
							end if
						else if allbutoneword == 8
							if nama == addone.word[2]
								set matchedname addone.word[2]
								goto matched
							else
								goto cknext
							end if
						else if allbutoneword == 9
							if nama == addone.word[3]
								set matchedname addone.word[3]
								goto matched
							else
								goto cknext
							end if
						else
							goto cknext
						end if
			else
				message "failed " secondlastletternum " " secondlastword " " logck.word[secondlastword] " " logck.word[secondlastword].letter[secondlastletternum] ""
					goto end
			end if
		label matched
		// becuase the the whole thing is saved as a "string" IE its saves as word(s)(yes even the numbers)
		// we have pull out the number, assing it to a new temp save, add to it, resave it to the whole "string".
			//message "matched: " nama " with: " matchedname ""
				if allbutoneword == 6
					if addone.word[0] == nama
						//message "Confirmed name match"
							set killnum addone.word[1]
							set killnum + 1
								setglobal dataseat addone.word[0]
								setglobal dataseat + " "
								setglobal dataseat + killnum
								setglobal dataseat + " "
								setglobal dataseat + addone.word[2]
									call updater
					else
						message "how in the chicken duck. 6"
					end if
				else if allbutoneword == 7
					if addone.word[1] == nama
						set killnum addone.word[2]
						set killnum + 1
							setglobal dataseat addone.word[0]
							setglobal dataseat + " "
							setglobal dataseat + addone.word[1]
							setglobal dataseat + " "
							setglobal dataseat + killnum
							setglobal dataseat + " "
							setglobal dataseat + addone.word[3]
								call updater
					else
						message "how in the chicken duck. 7"
					end if
				else if allbutoneword == 8
					if addone.word[2] == nama
						set killnum addone.word[3]
						set killnum + 1
							setglobal dataseat addone.word[0]
							setglobal dataseat + " "
							setglobal dataseat + addone.word[1]
							setglobal dataseat + " "
							setglobal dataseat + addone.word[2]
							setglobal dataseat + " "
							setglobal dataseat + killnum
							setglobal dataseat + " "
							setglobal dataseat + addone.word[4]
								call updater
					else
						message "how in the chicken duck. 8"
					end if
				else if allbutoneword == 9
					if addone.word[3] == nama
						set killnum addone.word[4]
						set killnum + 1
							setglobal dataseat addone.word[0]
							setglobal dataseat + " "
							setglobal dataseat + addone.word[1]
							setglobal dataseat + " "
							setglobal dataseat + addone.word[2]
							setglobal dataseat + " "
							setglobal dataseat + addone.word[3]
							setglobal dataseat + " "
							setglobal dataseat + killnum
							setglobal dataseat + " "
							setglobal dataseat + addone.word[5]
								call updater
					else
						message "how in the chicken duck. 9"
					end if
				else
					message "we dun fudged up som'tink"
				end if
	else
		goto cknext
else
	label cknext
	//message "no match; next!" // Very spammy dev message
	if nameck > 4
		goto end
	end if
	set nameck + 1
		goto top
end if
label end
	//message "done"
}

{

label reload
// the extra "\" here is required so selectitem command know it use both words
// normaly excudes anything after first word
"/selectitem \"whatzit <#" cknum "\" \r"
	pause 2
		if logck >= "no item"
			message "* no whatzit found skipping"
				goto skipper
		else if @my.selected_item == "Nothing"
			message "* something went wrong with item selection"
				goto ender
		else
			message "* loading " @my.selected_item ""
				call loader
					label skipper
					setglobal cknum + 1
						if cknum < 5
							goto reload
						else
							setglobal cknum 1
								call messwatch
						end if
		end if
label ender
}

adder
{
// check how many words there are, helps to know how many words the creature name is.
// we also have to check for time stamps for this to work. for now we aussume timestamps are on.
// we have to remove the "." in the dot in the name also for textlog scanning
set wordcount savedlog.num_words
if wordcount == ts1 // 9 words
	setglobal badnam savedlog.word[9]
		call scramble
			setglobal newsave nama
else if wordcount == ts2 // 10 words
	setglobal badnam savedlog.word[10]
		call scramble
			setglobal newsave savedlog.word[9]
			setglobal newsave + " "
			setglobal newsave + nama

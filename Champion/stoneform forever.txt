// real rock'n roller by Magnic 1.4 (untested)
// toggleable stoneform
// F1 on and off.


SETGLOBAL rockon 0

F1
{
if rockon == 0
	SETGLOBAL rockon 1
		if @my.waist_item != Stonegirdle
			"/equip Stonegirdle \r"
		end if
			message "* Lets ROCK!"
				"/useitem waist \r"
					goto rocktime
else 
	if rockon == 1
		SETGLOBAL rockon 0
			"/unequip waist \r"
				goto end
	end if
end if
label rocktime
	pause 900
		"/money \r"
			if rockon == 1
				pause 516 //<<--- ajust this number ---
			else
				goto end
			end if
			message "* rock n' rolling again"
				"/pose stand \r"
					"/useitem waist \r"
					if rockon == 1
						goto rocktime
					end if
end if
label end
}

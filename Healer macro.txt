// Healer macro by Magnic 1.0

// Right click = heal that person (eqiups healer item first)
// right clicking on no one or on the ground stops healing (this will also eqiup healer item)
// right click your self equips your moonstone
// no right click command will work with chain equiped

// "/hi" will set your right hand item as the healer item this macro will use (this is also done automadicly at log in)
// F1 = Radium healing

// F3 = heal pet
// F4 = pet command Attack
// F5 = pet command Stay
// F6 = pet command Obey

// F9 = pet command Go home
// F10 = Purgatory Pendant


SETGLOBAL Hchk 1
SETGLOBAL Right_H ""
SETGLOBAL Rad 1

@login
{
call sethealeritem
}

"/hi" // incase you where not holding your main healing item when you loged in.
{
call sethealeritem
}

sethealeritem
{
setglobal Right_H @my.right_item
message "* Healer Item set to" Right_H ""
if Right_H >= asklepian
	SETGLOBAL Rad 1
end if
}


click2
{
$any_click
set click2ck @clicksplayer
if @my.right_item >= "chain"
	message "* Disabled chain Equiped !" 
		goto end
else
	if click2ck != ""
		if @my.right_item != Right_H
			"/equip " Right_H "\r" 
	end if
		"/use " click2ck " \r"
			message "* Healing" click2ck ""
	else 
	if @my.right_item != Right_H
		"/equip " Right_H "\r" 
	end if
		"/use /off \r"
	end if
	if click2ck == @my.simple_name
		"/equip moonstone\r"
	end if
label end
	end if
end if
}

F1
{
if @my.right_item >= "chain"
	message "* Disabled chain Equiped !" 
		goto end
else
	if Rad == 1
		if @my.right_item != Right_H
			"/equip " Right_H "\r"
		end if
	"/use \r"
		message "*Radium Mode"
	else
		message "* your Healer Item isnt capble of that"
	end if
label end
end if
}

F3
{
if @my.right_item >= "chain"
	message "* Disabled chain Equiped !" 
		goto end
else
if @my.right_item != Right_H
"/equip " Right_H "\r"
end if
"/use /pet \r"
label end
end if
}

endr
{
if healname == nar
	setglobal numHn[numberply] ""
		message "Removed" nar "from Numpad" numberply ""
			setglobal nar ""
				goto exit
end if
"/select \r"
	"/select " healname "\r"
		setglobal nar @selplayer.name
			"/select \r"
				message "*" nar "is already on Numpad" numberply ""
					message "Click them again to remove name"
						label exit
							setglobal failname 1
}

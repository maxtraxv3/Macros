nameck
{
	if numHn[0] == healname
		setglobal numberply 0
			call endr
	else if numHn[1] == healname
		setglobal numberply 1
			call endr
	else if numHn[2] == healname
		setglobal numberply 2
			call endr
	else if numHn[3] == healname
		setglobal numberply 3
			call endr
	else if numHn[4] == healname
		setglobal numberply 4
			call endr
	else if numHn[5] == healname
		setglobal numberply 5
			call endr
	else if numHn[6] == healname
		setglobal numberply 6
			call endr
	else if numHn[7] == healname
		setglobal numberply 7
			call endr
	else if numHn[8] == healname
		setglobal numberply 8
			call endr
	else if numHn[9] == healname
		setglobal numberply 9
			call endr
	end if
}

endr
{
if healname == nar
	setglobal numHn[numberply] ""
		message "Removed" nar "from Numpad" numberply ""
			setglobal nar ""
				goto exit
end if
"/select \r"
	"/select " healname "\r"
		setglobal nar @selplayer.name
			"/select \r"
				message "*" nar "is already on Numpad" numberply ""
					message "Click them again to remove name"
						label exit
							setglobal failname 1
}

					"/use " numHn[0] " \r"

"/wn"
{
set nameck 0
message "* Numpad names:"
label top
if nameck < 9
	if numHn[nameck] != "Nothing"
		if numHn[nameck] != ""
			message "" numHn[nameck] "Numbpad" nameck ""
				set nameck + 1
		end if
	end if
goto top
else
	goto end
end if
label end
message "* ------end------ *"
}
